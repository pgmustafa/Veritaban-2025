BEGIN

DECLARE @NAMESURNAME NVARCHAR(100);
DECLARE @CITY NVARCHAR(100);
DECLARE @COUNTRY NVARCHAR(100);
DECLARE @ADDRESSTEXT NVARCHAR(100);
DECLARE @RAND INT;

SET @RAND = RAND() * 1000

SELECT 
  @NAMESURNAME = USERS.NAMESURNAME,
  @CITY = CITIES.CITY,
  @ADDRESSTEXT = a.ADDRESSTEXT,
  @COUNTRY = COUNTRIES.COUNTRY
FROM ADDRESS a
JOIN USERS ON USERS.ID = a.USERID
JOIN COUNTRIES ON COUNTRIES.ID = a.COUNTRYID
JOIN CITIES ON CITIES.ID = a.CITYID
where a.ID=@RAND

set @RAND=@RAND
PRINT  @RAND
END
SELECT * FROM ADDRESS
